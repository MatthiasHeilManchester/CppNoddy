213  /// Find the maximum stored absolute value in the mesh for a
214  /// given variable -- no interpolation is used
215  /// \param var The variable index whose maximum is being asked for
216  /// \return The value of the maximum (abs value)
217  double max(unsigned var) {
218  double max(0.0);
219  // step through the nodes
220  for(unsigned nodex = 0; nodex < m_nx; ++nodex) {
221  for(unsigned nodey = 0; nodey < m_ny; ++nodey) {
222  if(std::abs(m_vars[(nodex * m_ny + nodey) * m_nv + var ]) > max) {
223  max = std::abs(m_vars[(nodex * m_ny + nodey) * m_nv + var ]);
224  }
225  }
226  }

230  /// Get a bilinearly interpolated value at a specified point
231  /// \param x Physical (unmapped) x-coordinate in the 2D mesh
232  /// \param y Physical (unmapped) y-coordinate in the 2D mesh
233  /// \return A vector of bilinearly interpolated values
234  DenseVector<_Type> get_interpolated_vars(const double& x, const double& y)
235  {
236  double compX = FnComp_X(x);
237  double compY = FnComp_Y(y);
238  const double tol(1.e-8);
239  // check start and end
240  if ((compX < m_compX[0] - tol) || (compX > m_compX[m_nx-1] + tol)) {
241  std::string problem;
242  problem = " The TwoD_Node_Mesh.get_interpolated_vars method has been called with \n";
243  problem += " an x coordinate that lies outside the mesh. \n";
244  throw ExceptionRuntime(problem);
245  }
246  // check start and end
247  if ((compY < m_compY[0] - tol) || (compY > m_compY[m_ny-1] + tol)) {
248  std::string problem;
249  problem = " The TwoD_Node_Mesh.get_interpolated_vars method has been called with \n";
250  problem += " a y coordinate that lies outside the mesh. \n";
251  throw ExceptionRuntime(problem);
252  }
253  int bottom_j(-1);
254  for (unsigned j = 0; j < m_ny-1; ++j) {
255  if ((compY >= m_compY[j] - tol) && (compY <= m_compY[j+1] + tol)) {
256  bottom_j = j;
257  }
258  }
259  if (bottom_j == -1) {
260  std::string problem;
261  problem = " The TwoD_Node_Mesh.get_interpolated_vars method is broken.\n";
262  throw ExceptionRuntime(problem);
263  }
264  //
265  OneD_Node_Mesh<_Type> bottom_row( m_compX, m_nv );
266  OneD_Node_Mesh<_Type> top_row( m_compX, m_nv );
267  for (std::size_t i = 0; i < m_nx; ++i ) {
268  bottom_row.set_nodes_vars(i, this -> get_nodes_vars(i, bottom_j));
269  top_row.set_nodes_vars(i, this -> get_nodes_vars(i, bottom_j+1));
270  }
271  const double compY1 = m_compY[ bottom_j ];
272  const double compY2 = m_compY[ bottom_j+1 ];
273  DenseVector<_Type> result =
274  top_row.get_interpolated_vars(compX)*(compY-compY1)/(compY2-compY1)
275  + bottom_row.get_interpolated_vars(compX)*(compY2-compY)/(compY2-compY1);
276  return result;
277  }
