51  template <typename _Type>
52  void TwoD_Mapped_Node_Mesh<_Type>::init_mapping() {
53  #ifdef DEBUG
54  std::cout << "[DEBUG] Physical domain is [" << m_left << "," << m_right
55  << "] x [" << m_bottom << "," << m_top << "]\n";
56  std::cout << "[DEBUG] Computational domain is ["
57  << FnComp_X(m_left) << "," << FnComp_X(m_right)
58  << "] x [" << FnComp_Y(m_bottom) << "," << FnComp_Y(m_top) << "]\n";
59  #endif
60  {
61  // a uniform mesh in the computational coordinates
62  double comp_left(FnComp_X(m_left));
63  double comp_right(FnComp_X(m_right));
64  const double comp_delta_x = (comp_right - comp_left) / (m_nx - 1);
65  for(std::size_t i = 0; i < m_nx; ++i) {
66  m_compX[i] = comp_left + comp_delta_x * i;
67  }
68  }
69  {
70  // a uniform mesh in the computational coordinates
71  double comp_bottom(FnComp_Y(m_bottom));
72  double comp_top(FnComp_Y(m_top));
73  const double comp_delta_y = (comp_top - comp_bottom) / (m_ny - 1);
74  for(std::size_t j = 0; j < m_ny; ++j) {
75  m_compY[j] = comp_bottom + comp_delta_y * j;
76  }
77  }
78 
79  // temporary fill to span the domain -- corrected below
80  m_X = Utility::uniform_node_vector(m_left,m_right,m_nx);
81  m_Y = Utility::uniform_node_vector(m_bottom,m_top,m_ny);
82 
83  // we now need the corresponding m_Y coordinates in the physical domain
84  {
85  for(unsigned j = 1; j < m_ny; ++j) {
86  // for each node in m_compY
87  unsigned kmin(0);
88  double min(99e9);
89  for(unsigned k = 0; k < m_ny; ++k) {
90  // find the y value that is closest to it
91  if(std::abs(FnComp_Y(m_Y[k]) - m_compY[j]) < min) {
92  min = std::abs(FnComp_Y(m_Y[k]) - m_compY[j]);
93  kmin = k;
94  }
95  }
96  double y = m_Y[kmin];
97  double delta = 1.e-8;
98  double correction = 1.0;
99  do {
100  double newY = FnComp_Y(y + delta) - m_compY[j];
101  double oldY = FnComp_Y(y) - m_compY[j];
102  double deriv = (newY-oldY)/delta;
103  correction = -oldY/deriv;
104  y += correction;
105  } while(fabs(correction) > 1.e-8);
106  m_Y[ j ] = y;
107  }
108  }
109  // we now need the corresponding m_X coordinates in the physical domain
110  {
111  for(unsigned i = 1; i < m_nx; ++i) {
112  // for each node in m_compY
113  unsigned kmin(0);
114  double min(99e9);
115  for(unsigned k = 0; k < m_ny; ++k) {
116  // find the y value that is closest to it
117  if(std::abs(FnComp_X(m_X[k]) - m_compX[i]) < min) {
118  min = std::abs(FnComp_X(m_X[k]) - m_compX[i]);
119  kmin = k;
120  }
121  }
122  double x = m_X[kmin];
123  double delta = 1.e-8;
124  double correction = 1.0;
125  do {
126  double newX = FnComp_X(x + delta) - m_compX[i];
127  double oldX = FnComp_X(x) - m_compX[i];
128  double deriv = (newX-oldX)/delta;
129  correction = -oldX/deriv;
130  x += correction;
131  } while(fabs(correction) > 1.e-8);
132  m_X[ i ] = x;
133  }
134  }
135  }

175  void TwoD_Mapped_Node_Mesh<D_complex>::normalise(const std::size_t& var) {
176  //std::cout << "[DEBUG] asked to normalise a complex mesh\n";
177  unsigned max_nx(0);
178  unsigned max_ny(0);
179  double max(0.0);
180  // step through the nodes
181  for(unsigned nodex = 0; nodex < m_X.size(); ++nodex) {
182  for(unsigned nodey = 0; nodey < m_Y.size(); ++nodey) {
183  if(std::abs(m_vars[(nodex * m_ny + nodey) * m_nv + var ]) > max) {
184  max = std::abs(m_vars[(nodex * m_ny + nodey) * m_nv + var ]);
185  max_nx = nodex;
186  max_ny = nodey;
187  }
188  }
189  }
190  D_complex factor(m_vars[(max_nx * m_ny + max_ny) * m_nv + var ]);
191  //std::cout << "[DEBUG] MAX |variable| had complex value of " << factor << "\n";
192  m_vars.scale(1./factor);
193  }
